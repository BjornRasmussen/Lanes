001package org.openstreetmap.josm.plugins.lanes;
002
003import org.openstreetmap.josm.gui.IconToggleButton;
004import org.openstreetmap.josm.gui.MainApplication;
005import org.openstreetmap.josm.gui.MapFrame;
006import org.openstreetmap.josm.plugins.Plugin;
007import org.openstreetmap.josm.plugins.PluginInformation;
008
009public class LanesPlugin extends Plugin {
010
011    public LanesPlugin(PluginInformation info) {
012        super(info);
013    }
014
015    @Override
016    public void mapFrameInitialized(MapFrame oldFrame, MapFrame newFrame) {
017        if (oldFrame == null && newFrame != null) {
018            MainApplication.getMap().addMapMode(new IconToggleButton(new LaneMappingMode(MainApplication.getMap())));
019        }
020    }
021}

001package org.openstreetmap.josm.plugins.lanes;
002
003import org.openstreetmap.josm.gui.MapView;
004
005import java.awt.*;
006
007public class RoadEdge extends RoadPiece {
008
009    protected RoadEdge(int direction, int position, MapView mv, RoadSegmentRenderer parent) {
010        super(direction, position, mv, parent);
011    }
012
013    @Override
014    double getWidth(boolean start) {
015        return Utils.RENDERING_WIDTH_DIVIDER;
016    }
017
018    @Override
019    void render(Graphics2D g) {
020        renderAsphalt(g);
021        renderRoadLine(g, _offsetStart, _offsetEnd, Utils.DividerType.SOLID, _way.isOneway() == 1 &&
022                (Utils.isRightHand(_way) ? _direction == -1 : _direction == 1));
023    }
024}

169    private void getLanesFromWay() {
170        Map<String, String> tags = _way.getInterestingTags();
171
172        if (_way.isOneway() == -1) {
173            _isValid = false;
174        }
175
176        int numLanesForward  = getLanesInDirectionFromSuffix(1);
177        int numLanesBackward = getLanesInDirectionFromSuffix(-1);
178        int numLanesBothWays = getLanesInDirectionFromSuffix(0);
179
180        if (tags.containsKey("lanes:forward") && numLanesForward == -1) {
181            numLanesForward = Integer.parseInt(tags.get("lanes:forward"));
182        }
183        if (tags.containsKey("lanes") && _way.isOneway() == 1 && numLanesForward == -1) {
184            numLanesForward = Integer.parseInt(tags.get("lanes"));
185        }
186        if (tags.containsKey("lanes:backward") && numLanesBackward == -1) {
187            numLanesBackward = Integer.parseInt(tags.get("lanes:backward"));
188        }
189        if (tags.containsKey("lanes:both_ways") && numLanesBothWays == -1) {
190            numLanesBothWays = Integer.parseInt(tags.get("lanes:both_ways"));
191        }
192
193        if (numLanesBothWays == -1) numLanesBothWays = 0; // Assume no centre lane.
194
195        if (tags.containsKey("lanes") && _way.isOneway() == 0) {
196            int lanes = Integer.parseInt(tags.get("lanes")) - numLanesBothWays;
197            if (numLanesForward == -1 && numLanesBackward == -1) {
198                numLanesForward = lanes-lanes/2;
199                numLanesBackward = lanes/2;
200            } else if (numLanesForward == -1) {
201                numLanesForward = lanes - numLanesBackward;
202            } else if (numLanesBackward == -1) {
203                numLanesBackward = lanes - numLanesForward;
204            }
205        }
206
207        _leftRoadEdge = new RoadEdge(isRightHand() ? -1 : 1, -1);
208        _rightRoadEdge = new RoadEdge(isRightHand() ? 1 : -1, -1);
209
210        for (int i = 0; i < numLanesForward; i++) {
211            _forwardLanes.add(new Lane(1, i));
212            if (i != numLanesForward-1) _forwardDividers.add(new Divider(1, i));
213        }
214        for (int i = 0; i < numLanesBackward; i++) {
215            _backwardLanes.add(new Lane(-1, i));
216            if (i != numLanesBackward-1) _backwardDividers.add(new Divider(-1, i));
217        }
218        if (numLanesBackward != 0) {
219            if (numLanesBothWays <= 0) {
220                _bothWaysLane = new Divider(0, 0);
221            } else {
222                _bothWaysLane = new Lane(0, 0);
223            }
224        }
225    }

227    private int getLanesInDirectionFromSuffix(int direction) {
228        Map<String, String> tags = _way.getInterestingTags();
229        int numLanes = -1;
230        for (String key : tags.keySet()) {
231            if (!key.contains("note") && (direction == 1 ? ((key.endsWith(":lanes") && _way.isOneway() == 1) || key.endsWith(":lanes:forward")) :
232                    direction == 0 ? key.endsWith(":lanes:both_ways") : key.endsWith(":lanes:backward"))) {
233
234                // This runs if the tag being analyzed is a lane tag applying to this direction.
235                int len = ("a" + tags.get(key) + "a").split("\\|").length;
236                if (numLanes == -1) numLanes = len;
237                if (numLanes != len) {
238                    _isValid = false;
239                }
240            }
241        }
242        return numLanes;
243    }

253    private double getPlacementAt(boolean start, boolean ignoreWidthTags) {
254        try {
255            // Get string representation of placement.
256            String placement = getPlacementTag(start);
257            int direction = placement.charAt(placement.length()-1) == 'f' ? 1 : placement.charAt(placement.length()-1) == 'b' ? -1 : 0;
258            placement = placement.substring(0, placement.length()-1);
259
260            String placementOther = getPlacementTag(!start);
261            int directionOther = -2;
262            if (placementOther != null) {
263                directionOther = placementOther.charAt(placementOther.length()-1) == 'f' ? 1 :
264                        placementOther.charAt(placementOther.length()-1) == 'b' ? -1 : 0;
265                placementOther = placementOther.substring(0, placementOther.length()-1);
266            }
267
268
269
270            // Get offset
271            List<RoadPiece> pieces = getRoadPieces(false);
272            int lane = Integer.parseInt(placement.split(":")[1]);
273            int laneOther = Integer.MIN_VALUE;
274            if (placementOther != null) Integer.parseInt(placementOther.split(":")[1]);
275            double offsetSoFar = 0;
276            boolean valid = false;
277
278            // For when placement is outside of the range of this road segment:
279            RoadPiece minForward = null;
280            RoadPiece maxForward = null;
281            RoadPiece minBackward = null;
282            RoadPiece maxBackward = null;
283
284            double minForwardOffset = Double.NaN;
285            double maxForwardOffset = Double.NaN;
286            double minBackwardOffset = Double.NaN;
287            double maxBackwardOffset = Double.NaN;
288
289            for (int i = 1; i < pieces.size(); i++) {
290                RoadPiece p = pieces.get(i);
291
292                boolean correctLane = p._position+1 == lane && p._direction == direction && (p._direction == 0 || p instanceof Lane);
293                boolean correctLaneOther = false;
294                if (placementOther != null) correctLaneOther = p._position+1 == laneOther && p._direction == directionOther && (p._direction == 0 || p instanceof Lane);
295
296                offsetSoFar += (ignoreWidthTags && !correctLane && !correctLaneOther) ? (pieces.get(i-1) instanceof Lane ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER) :
297                        RENDERING_WIDTH_DIVIDER)/2 : (pieces.get(i-1).getWidth(start) / 2);
298                offsetSoFar += (ignoreWidthTags && !correctLane && !correctLaneOther) ? (p instanceof Lane ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER) :
299                        RENDERING_WIDTH_DIVIDER)/2 : (p.getWidth(start) / 2);
300
301                if (p._direction == 1 && p instanceof Lane && (minForward == null || minForward._position > p._position)) {
302                    minForward = p;
303                    minForwardOffset = offsetSoFar;
304                }
305                if (p._direction == 1 && p instanceof Lane && (maxForward == null || maxForward._position < p._position)) {
306                    maxForward = p;
307                    maxForwardOffset = offsetSoFar;
308                }
309                if (p._direction == -1 && p instanceof Lane  && (minBackward == null || minBackward._position > p._position)) {
310                    minBackward = p;
311                    minBackwardOffset = offsetSoFar;
312                }
313                if (p._direction == -1 && p instanceof Lane  && (maxBackward == null || maxBackward._position < p._position)) {
314                    maxBackward = p;
315                    maxBackwardOffset = offsetSoFar;
316                }
317
318                if (p._position+1 == lane && p._direction == direction && (p._direction == 0 || p instanceof Lane)) {
319                    if ((direction == 1 && placement.startsWith("left_of")) ||
320                            (direction == -1 && placement.startsWith("right_of"))) {
321                        offsetSoFar -= RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : (p.getWidth(start) / 2));
322                    }
323                    if ((direction == 1 && placement.startsWith("right_of")) ||
324                            (direction == -1 && placement.startsWith("left_of"))) {
325                        offsetSoFar += RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : (p.getWidth(start) / 2));
326                    }
327                    valid = true;
328                    break;
329                }
330            }
331
332            if (!valid && direction == 1) {
333                if (minForward._position+1 > lane) {
334                    int numLanesAway = minForward._position + 1 - lane;
335                    offsetSoFar = minForwardOffset - numLanesAway * WIDTH_LANES;
336
337                    if (placement.startsWith("left_of")) {
338                        offsetSoFar -= RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : minForward.getWidth(start) / 2);
339                    }
340                    if (placement.startsWith("right_of")) {
341                        offsetSoFar += RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : minForward.getWidth(start) / 2);
342                    }
343                }
344                if (maxForward._position+1 < lane) {
345                    int numLanesAway = maxForward._position + 1 - lane;
346                    offsetSoFar = maxForwardOffset - numLanesAway * WIDTH_LANES;
347
348                    if (placement.startsWith("left_of")) {
349                        offsetSoFar -= RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : maxForward.getWidth(start) / 2);
350                    }
351                    if (placement.startsWith("right_of")) {
352                        offsetSoFar += RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : maxForward.getWidth(start) / 2);
353                    }
354                }
355                valid = true;
356            }
357
358            if (!valid && direction == -1) {
359                if (minBackward._position+1 > lane) {
360                    int numLanesAway = minBackward._position + 1 - lane;
361                    offsetSoFar = minBackwardOffset + numLanesAway * WIDTH_LANES;
362
363                    if (placement.startsWith("left_of")) {
364                        offsetSoFar += RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : minBackward.getWidth(start) / 2);
365                    }
366                    if (placement.startsWith("right_of")) {
367                        offsetSoFar -= RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : minBackward.getWidth(start) / 2);
368                    }
369                }
370                if (maxBackward._position+1 < lane) {
371                    int numLanesAway = maxBackward._position + 1 - lane;
372                    offsetSoFar = maxBackwardOffset + numLanesAway * WIDTH_LANES;
373
374                    if (placement.startsWith("left_of")) {
375                        offsetSoFar -= RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : maxBackward.getWidth(start) / 2);
376                    }
377                    if (placement.startsWith("right_of")) {
378                        offsetSoFar += RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (WIDTH_LANES-RENDERING_WIDTH_DIVIDER)/2 : maxBackward.getWidth(start) / 2);
379                    }
380                }
381                valid = true;
382            }
383
384            if (!valid) return getWidth(start) / 2.0;
385
386            return offsetSoFar;
387        } catch (Exception e) {
388            return getWidth(start) / 2.0;
389        }
390    }

430    private List<RoadPiece> getRoadPieces(boolean renderingOrder) {
431        List<RoadPiece> output = new ArrayList<>();
432
433        if (renderingOrder) {
434            output.addAll(getLanesAndDividers(_forwardLanes, _forwardDividers, 1));
435            output.addAll(getLanesAndDividers(_backwardLanes, _backwardDividers, -1));
436            if (_backwardLanes.size() != 0 || (_bothWaysLane instanceof Lane)) {
437                output.add(_bothWaysLane);
438            }
439            output.add(_leftRoadEdge);
440            output.add(_rightRoadEdge);
441        } else {
442            output.add(_leftRoadEdge);
443            if (isRightHand()) {
444                output.addAll(getLanesAndDividers(_backwardLanes, _backwardDividers, -1));
445                if (_backwardLanes.size() != 0 || _bothWaysLane instanceof Lane) output.add(_bothWaysLane);
446                output.addAll(getLanesAndDividers(_forwardLanes, _forwardDividers, 1));
447            } else {
448                output.addAll(getLanesAndDividers(_forwardLanes, _forwardDividers, 1));
449                if (_backwardLanes.size() != 0 || _bothWaysLane instanceof Lane) output.add(_bothWaysLane);
450                output.addAll(getLanesAndDividers(_backwardLanes, _backwardDividers, -1));
451            }
452            output.add(_rightRoadEdge);
453        }
454        return output;
455    }

457    private List<RoadPiece> getLanesAndDividers(List<RoadPiece> lanes, List<RoadPiece> dividers, int direction) {
458        List<RoadPiece> output = new ArrayList<>();
459        if (direction > 0) {
460            for (int i = 0; i < lanes.size(); i++) {
461                output.add(lanes.get(i));
462                if (i < lanes.size() - 1) output.add(dividers.get(i));
463            }
464        } else {
465            for (int i = lanes.size()-1; i >= 0; i--) {
466                output.add(lanes.get(i));
467                if (i != 0) output.add(dividers.get(i-1));
468            }
469        }
470        return output;
471    }

477    private void getOffsets() {
478        List<RoadPiece> roadPieces = getRoadPieces(false);
479        double offsetSoFarStart = _offsetToLeftStart;
480        double offsetSoFarEnd = _offsetToLeftEnd;
481        for (int i = 0; i < roadPieces.size(); i++) {
482            roadPieces.get(i).setOffset(offsetSoFarStart, offsetSoFarEnd);
483            offsetSoFarStart -= roadPieces.get(i).getWidth(true)/2 +
484                    (i != roadPieces.size()-1 ? roadPieces.get(i+1).getWidth(true)/2 : 0);
485            offsetSoFarEnd -= roadPieces.get(i).getWidth(false)/2 +
486                    (i != roadPieces.size()-1 ? roadPieces.get(i+1).getWidth(false)/2 : 0);
487        }
488    }

494    private Way getParallel(Way way, double offsetStart, double offsetEnd, boolean useAngleOffset) {
495        LatLon[] points = new LatLon[way.getNodesCount()];
496        double[] distanceIntoWay = new double[way.getNodesCount()];
497        double distanceOfWay = 0;
498        for (int i = 0; i < points.length; i++) {
499            points[i] = way.getNode(i).getCoor();
500            if (i != 0) {
501                try {
502                    distanceOfWay += points[i - 1].greatCircleDistance(points[i]);
503                } catch (Exception ignored) {}
504            }
505            distanceIntoWay[i] = distanceOfWay;
506        }
507
508        // Get angle offset:
509        double angleOffset = (useAngleOffset ? -1 : 0) * Math.asin((offsetEnd-offsetStart)/distanceOfWay);
510
511        LatLon[] output = new LatLon[points.length];
512
513        // Deal with first node
514        double angle;
515        if (points.length < 2) return null;
516        try {
517            // If points are at same location, return the way, since getting a parallel way would be hard.
518            angle = points[0].bearing(points[1]);
519        } catch (NullPointerException e) {
520            return way;
521        }
522        output[0] = getLatLonRelative(points[0], angle-(Math.PI / 2.0) + angleOffset, offsetStart);
523
524        // Deal with all other nodes
525        for (int i = 1; i < points.length - 1; i++) {
526            double angleToPrevPoint;
527            double angleToNextPoint;
528            try {
529                // If points are at same location, return the way, since getting a parallel way would be hard.s
530                angleToPrevPoint = points[i].bearing(points[i - 1]);
531                angleToNextPoint = points[i].bearing(points[i + 1]);
532            } catch (NullPointerException e) {
533                return way;
534            }
535            double angleBetween = (angleToNextPoint + angleToPrevPoint) / 2;
536            if (angleToNextPoint < angleToPrevPoint) angleBetween = (angleBetween + Math.PI) % (Math.PI * 2.0);
537
538            double amountThrough = distanceIntoWay[i]/distanceOfWay;
539            double offsetAtNode = offsetStart * (1 - amountThrough) + offsetEnd * amountThrough;
540
541            double anglePrevToNormal = (angleBetween - angleToNextPoint) % (2 * Math.PI);
542
543            double offset = offsetAtNode / Math.abs(Math.sin(anglePrevToNormal));
544
545            output[i] = getLatLonRelative(points[i], angleBetween + angleOffset, offset);
546        }
547
548        // Deal with last node
549        double angleToPrev = points[points.length-1].bearing(points[points.length-2]);
550        output[points.length-1] = getLatLonRelative(points[points.length-1], angleToPrev+(Math.PI / 2.0) + angleOffset, offsetEnd);
551        // TODO handle loop roads.
552
553        // Convert to way format and return
554        List<Node> outputNodes = new ArrayList<>();
555        for (LatLon ll : output) outputNodes.add(new Node(ll));
556
557        Way outputWay = new Way();
558        outputWay.setNodes(outputNodes);
559
560        return outputWay;
561    }

563    private LatLon getLatLonRelative(LatLon from, double bearing, double numMeters) {
564        double metersPerDegreeLat = 111319.5;
565        double metersPerDegreeLon = metersPerDegreeLat * Math.cos(from.getY() / 180.0 * Math.PI);
566        double dx = Math.sin(bearing) * numMeters / metersPerDegreeLon;
567        double dy = Math.cos(bearing) * numMeters / metersPerDegreeLat;
568        return new LatLon(from.lat() + dy, from.lon() + dx);
569    }

723        protected double parseLaneWidth(String value) {
724            if (value.endsWith(" lane")) {
725                return WIDTH_LANES*Double.parseDouble(value.substring(0, value.length()-5));
726            } else if (value.endsWith(" lanes")) {
727                return WIDTH_LANES*Double.parseDouble(value.substring(0, value.length()-6));
728            } else if (value.endsWith(" m")) {
729                return Double.parseDouble(value.substring(0, value.length()-2));
730            } else if (value.endsWith(" km")) {
731                return 1000 * Double.parseDouble(value.substring(0, value.length()-3));
732            } else if (value.endsWith(" mi")) {
733                return 1609.344 * Double.parseDouble(value.substring(0, value.length()-3));
734            } else if (value.endsWith("\'")) {
735                return 1 / 3.28084 * Double.parseDouble(value.substring(0, value.length()-1));
736            } else if (value.endsWith("\"") && !value.contains("'")) {
737                return 1 / 3.28084 / 12 * Double.parseDouble(value.substring(0, value.length()-1));
738            } else if (value.endsWith("\"") && value.contains("'")) {
739                String[] split = value.split("'");
740                double feet = Double.parseDouble(split[0]);
741                double inches = Double.parseDouble(split[1].substring(0, value.length()-1));
742                return 1 / 3.28084 * (feet + inches/12);
743            }
744            return Double.parseDouble(value);
745        }

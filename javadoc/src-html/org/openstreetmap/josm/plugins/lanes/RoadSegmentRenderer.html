053
054        if (!_isValid) {
055            // Get the centre line of the road to be rendered.
056            int[] pointXs = new int[_way.getNodesCount()];
057            int[] pointYs = new int[_way.getNodesCount()];
058            for (int i = 0; i < _way.getNodesCount(); i++) {
059                pointXs[i] = _mv.getPoint(_way.getNode(i).getCoor()).x;
060                pointYs[i] = _mv.getPoint(_way.getNode(i).getCoor()).y;
061            }
062
063            // Set the color and width to the "invalid" defaults.
064            g.setColor(Utils.DEFAULT_INVALID_COLOR);
065            g.setStroke(new BasicStroke((int) (Utils.
066                    WIDTH_INVALID_METERS * 100.0 / _mv.getDist100Pixel()),
067                    BasicStroke.CAP_BUTT, BasicStroke.JOIN_ROUND));
068
069            // Draw the way
070            g.drawPolyline(pointXs, pointYs, pointXs.length);
071
072            // Get rid of that white rectangle that was appearing around the screen at high zoom levels:
073            g.setColor(new Color(0, 0, 0, 0));
074            g.setStroke(GuiHelper.getCustomizedStroke("0"));
075            return;
076        }

111    public void updateChangeTags() {
112        // This method completely rebuilds the change tags based on the info stored in the child RoadPieces.
113        List<Command> cmds = new LinkedList<>();
114
115        // Set up change lists for each direction:
116        String[] forward = new String[_forwardLanes.size()];
117        String[] backward = new String[_backwardLanes.size()];
118        for (RoadPiece p : getRoadPieces(false)) {
119            if (p instanceof Lane) {
120                if (p._direction == 1) {
121                    forward[p._position] = ((Lane) p).getChange();
122                }
123                if (p._direction == -1) {
124                    backward[p._position] = ((Lane) p).getChange();
125                }
126            }
127        }
128
129        // Convert forward to string.
130        boolean forwardIsAllSame = true;
131        String forwardChange = "";
132        for (int i = 0; i < forward.length; i++) {
133            if (i != 0 && !forward[i-1].equals(forward[i])) forwardIsAllSame = false;
134            if (i != 0) forwardChange += "|";
135            forwardChange += forward[i];
136        }
137
138        // Convert backward to string.
139        boolean backwardIsAllSame = true;
140        String backwardChange = "";
141        for (int i = 0; i < backward.length; i++) {
142            if (i != 0 && !backward[i-1].equals(backward[i])) backwardIsAllSame = false;
143            if (i != 0) backwardChange += "|";
144            backwardChange += backward[i];
145        }
146
147        if (forwardIsAllSame) forwardChange = forwardChange.split("\\|")[0];
148        if (backwardIsAllSame) backwardChange = backwardChange.split("\\|")[0];
149
150        removeTag(_way, "change", cmds);
151        removeTag(_way, "change:forward", cmds);
152        removeTag(_way, "change:backward", cmds);
153        removeTag(_way, "change:lanes", cmds);
154        removeTag(_way, "change:lanes:forward", cmds);
155        removeTag(_way, "change:lanes:backward", cmds);
156
157        if (forwardIsAllSame && backwardIsAllSame && forwardChange.equals(backwardChange) && !forwardChange.equals("")) {
158            addTag(_way, "change", forwardChange, cmds);
159        }
160
161        if (forwardIsAllSame && !(backwardIsAllSame && forwardChange.equals(backwardChange)) && !forwardChange.equals("")) {
162            addTag(_way, "change" + (_way.isOneway() == 1 ? "" : ":forward"), forwardChange, cmds);
163
164        }
165
166        if (backwardIsAllSame && !(forwardIsAllSame && forwardChange.equals(backwardChange)) && !forwardChange.equals("")) {
167            addTag(_way, "change:backward", backwardChange, cmds);
168        }
169
170        if (!forwardIsAllSame && !forwardChange.equals("")) {
171            addTag(_way, "change:lanes" + (_way.isOneway() == 1 ? "" : ":forward"), forwardChange, cmds);
172        }
173
174        if (!backwardIsAllSame && !backwardChange.equals("")) {
175            addTag(_way, "change:lanes:backward", backwardChange, cmds);
176        }
177
178        SequenceCommand c = new SequenceCommand("Change Divider Type", cmds);
179        UndoRedoHandler.getInstance().add(c);
180    }

182    protected void removeTag(Way way, String key, java.util.List<Command> cmds) {
183        addTag(way, key, null, cmds);
184    }

186    protected void addTag(Way way, String key, String value, List<Command> cmds) {
187        Map<String, String> tags = way.getInterestingTags();
188        if (value == null) {
189            tags.remove(key);
190        } else {
191            tags.put(key, value);
192        }
193        Way way2 = new Way();
194        way2.cloneFrom(way);
195        way2.setKeys(tags);
196        cmds.add(new ChangeCommand(way, way2));
197    }

203    private void getLanesFromWay() {
204        Map<String, String> tags = _way.getInterestingTags();
205
206        if (_way.isOneway() == -1) {
207            _isValid = false;
208        }
209
210        int numLanesForward  = getLanesInDirectionFromSuffix(1);
211        int numLanesBackward = getLanesInDirectionFromSuffix(-1);
212        int numLanesBothWays = getLanesInDirectionFromSuffix(0);
213
214        if (tags.containsKey("lanes:forward") && numLanesForward == -1) {
215            numLanesForward = Integer.parseInt(tags.get("lanes:forward"));
216        }
217        if (tags.containsKey("lanes") && _way.isOneway() == 1 && numLanesForward == -1) {
218            numLanesForward = Integer.parseInt(tags.get("lanes"));
219        }
220        if (tags.containsKey("lanes:backward") && numLanesBackward == -1) {
221            numLanesBackward = Integer.parseInt(tags.get("lanes:backward"));
222        }
223        if (tags.containsKey("lanes:both_ways") && numLanesBothWays == -1) {
224            numLanesBothWays = Integer.parseInt(tags.get("lanes:both_ways"));
225        }
226
227        if (numLanesBothWays == -1) numLanesBothWays = 0; // Assume no centre lane.
228
229        if (tags.containsKey("lanes") && _way.isOneway() == 0) {
230            int lanes = Integer.parseInt(tags.get("lanes")) - numLanesBothWays;
231            if (numLanesForward == -1 && numLanesBackward == -1) {
232                numLanesForward = lanes-lanes/2;
233                numLanesBackward = lanes/2;
234            } else if (numLanesForward == -1) {
235                numLanesForward = lanes - numLanesBackward;
236            } else if (numLanesBackward == -1) {
237                numLanesBackward = lanes - numLanesForward;
238            }
239        }
240
241        _leftRoadEdge = new RoadEdge(Utils.isRightHand(_way) ? -1 : 1, -1, _mv, this);
242        _rightRoadEdge = new RoadEdge(Utils.isRightHand(_way) ? 1 : -1, -1, _mv, this);
243
244        for (int i = 0; i < numLanesForward; i++) {
245            _forwardLanes.add(new Lane(1, i, _mv, this));
246            if (i != numLanesForward-1) {
247                _forwardDividers.add(new Divider(1, i, _mv, this));
248                _forwardLanes.get(i).setRightPiece(_forwardDividers.get(i));
249                _forwardDividers.get(i).setLeftPiece(_forwardLanes.get(i));
250            }
251            if (i != 0) {
252                _forwardLanes.get(i).setLeftPiece(_forwardDividers.get(i-1));
253                _forwardDividers.get(i-1).setRightPiece(_forwardLanes.get(i));
254            }
255        }
256        for (int i = 0; i < numLanesBackward; i++) {
257            _backwardLanes.add(new Lane(-1, i, _mv, this));
258            if (i != numLanesBackward-1) {
259                _backwardDividers.add(new Divider(-1, i, _mv, this));
260                _backwardLanes.get(i).setRightPiece(_backwardDividers.get(i));
261                _backwardDividers.get(i).setLeftPiece(_backwardLanes.get(i));
262            }
263            if (i != 0) {
264                _backwardLanes.get(i).setLeftPiece(_backwardDividers.get(i-1));
265                _backwardDividers.get(i-1).setRightPiece(_backwardLanes.get(i));
266            }
267        }
268        if (numLanesBackward != 0) {
269            if (numLanesBothWays <= 0) {
270                _bothWaysLane = new Divider(0, 0, _mv, this);
271            } else {
272                _bothWaysLane = new Lane(0, 0, _mv, this);
273            }
274        }
275        if (_backwardLanes.size() != 0) {
276            if (Utils.isRightHand(_way)) {
277                _backwardLanes.get(0).setLeftPiece(_bothWaysLane);
278                _bothWaysLane.setLeftPiece(_backwardLanes.get(0));
279                _forwardLanes.get(0).setLeftPiece(_bothWaysLane);
280                _bothWaysLane.setRightPiece(_forwardLanes.get(0));
281            } else {
282                _backwardLanes.get(_backwardLanes.size() - 1).setRightPiece(_bothWaysLane);
283                _bothWaysLane.setRightPiece(_backwardLanes.get(_backwardLanes.size() - 1));
284                _forwardLanes.get(_forwardLanes.size() - 1).setRightPiece(_bothWaysLane);
285                _bothWaysLane.setLeftPiece(_forwardLanes.get(_forwardLanes.size() - 1));
286            }
287        }
288    }

290    private int getLanesInDirectionFromSuffix(int direction) {
291        Map<String, String> tags = _way.getInterestingTags();
292        int numLanes = -1;
293        for (String key : tags.keySet()) {
294            if (!key.contains("note") && (direction == 1 ? ((key.endsWith(":lanes") && _way.isOneway() == 1) || key.endsWith(":lanes:forward")) :
295                    direction == 0 ? key.endsWith(":lanes:both_ways") : key.endsWith(":lanes:backward"))) {
296
297                // This runs if the tag being analyzed is a lane tag applying to this direction.
298                int len = ("a" + tags.get(key) + "a").split("\\|").length;
299                if (numLanes == -1) numLanes = len;
300                if (numLanes != len) {
301                    _isValid = false;
302                }
303            }
304        }
305        return numLanes;
306    }

308    private void getPlacementInformation() {
309        _offsetToLeftStart = getPlacementAt(true, false);
310        _offsetToLeftEnd = getPlacementAt(false, false);
311        double placementDiff = getPlacementAt(false, true) - getPlacementAt(true, true);
312        _alignment = Utils.getParallel(_way, 0, placementDiff, true);
313        _offsetToLeftEnd -= placementDiff;
314    }

316    private double getPlacementAt(boolean start, boolean ignoreWidthTags) {
317        try {
318            // Get string representation of placement.
319            String placement = getPlacementTag(start);
320            int direction = placement.charAt(placement.length()-1) == 'f' ? 1 : placement.charAt(placement.length()-1) == 'b' ? -1 : 0;
321            placement = placement.substring(0, placement.length()-1);
322
323            String placementOther = getPlacementTag(!start);
324            int directionOther = -2;
325            if (placementOther != null) {
326                directionOther = placementOther.charAt(placementOther.length()-1) == 'f' ? 1 :
327                        placementOther.charAt(placementOther.length()-1) == 'b' ? -1 : 0;
328                placementOther = placementOther.substring(0, placementOther.length()-1);
329            }
330
331
332
333            // Get offset
334            List<RoadPiece> pieces = getRoadPieces(false);
335            int lane = Integer.parseInt(placement.split(":")[1]);
336            int laneOther = Integer.MIN_VALUE;
337            if (placementOther != null) Integer.parseInt(placementOther.split(":")[1]);
338            double offsetSoFar = 0;
339            boolean valid = false;
340
341            // For when placement is outside of the range of this road segment:
342            RoadPiece minForward = null;
343            RoadPiece maxForward = null;
344            RoadPiece minBackward = null;
345            RoadPiece maxBackward = null;
346
347            double minForwardOffset = Double.NaN;
348            double maxForwardOffset = Double.NaN;
349            double minBackwardOffset = Double.NaN;
350            double maxBackwardOffset = Double.NaN;
351
352            for (int i = 1; i < pieces.size(); i++) {
353                RoadPiece p = pieces.get(i);
354
355                boolean correctLane = p._position+1 == lane && p._direction == direction && (p._direction == 0 || p instanceof Lane);
356                boolean correctLaneOther = false;
357                if (placementOther != null) correctLaneOther = p._position+1 == laneOther && p._direction == directionOther && (p._direction == 0 || p instanceof Lane);
358
359                offsetSoFar += (ignoreWidthTags && !correctLane && !correctLaneOther) ? (pieces.get(i-1) instanceof Lane ? (Utils.
360                        WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER) :
361                        Utils.RENDERING_WIDTH_DIVIDER)/2 : (pieces.get(i-1).getWidth(start) / 2);
362                offsetSoFar += (ignoreWidthTags && !correctLane && !correctLaneOther) ? (p instanceof Lane ? (Utils.
363                        WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER) :
364                        Utils.RENDERING_WIDTH_DIVIDER)/2 : (p.getWidth(start) / 2);
365
366                if (p._direction == 1 && p instanceof Lane && (minForward == null || minForward._position > p._position)) {
367                    minForward = p;
368                    minForwardOffset = offsetSoFar;
369                }
370                if (p._direction == 1 && p instanceof Lane && (maxForward == null || maxForward._position < p._position)) {
371                    maxForward = p;
372                    maxForwardOffset = offsetSoFar;
373                }
374                if (p._direction == -1 && p instanceof Lane  && (minBackward == null || minBackward._position > p._position)) {
375                    minBackward = p;
376                    minBackwardOffset = offsetSoFar;
377                }
378                if (p._direction == -1 && p instanceof Lane  && (maxBackward == null || maxBackward._position < p._position)) {
379                    maxBackward = p;
380                    maxBackwardOffset = offsetSoFar;
381                }
382
383                if (p._position+1 == lane && p._direction == direction && (p._direction == 0 || p instanceof Lane)) {
384                    if ((direction == 1 && placement.startsWith("left_of")) ||
385                            (direction == -1 && placement.startsWith("right_of"))) {
386                        offsetSoFar -= Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (Utils.
387                                WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : (p.getWidth(start) / 2));
388                    }
389                    if ((direction == 1 && placement.startsWith("right_of")) ||
390                            (direction == -1 && placement.startsWith("left_of"))) {
391                        offsetSoFar += Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (Utils.
392                                WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : (p.getWidth(start) / 2));
393                    }
394                    valid = true;
395                    break;
396                }
397            }
398
399            if (!valid && direction == 1) {
400                if (minForward._position+1 > lane) {
401                    int numLanesAway = minForward._position + 1 - lane;
402                    offsetSoFar = minForwardOffset - numLanesAway * Utils.WIDTH_LANES;
403
404                    if (placement.startsWith("left_of")) {
405                        offsetSoFar -= Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (Utils.
406                                WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : minForward.getWidth(start) / 2);
407                    }
408                    if (placement.startsWith("right_of")) {
409                        offsetSoFar += Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (Utils.
410                                WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : minForward.getWidth(start) / 2);
411                    }
412                }
413                if (maxForward._position+1 < lane) {
414                    int numLanesAway = maxForward._position + 1 - lane;
415                    offsetSoFar = maxForwardOffset - numLanesAway * Utils.WIDTH_LANES;
416
417                    if (placement.startsWith("left_of")) {
418                        offsetSoFar -= Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (Utils.
419                                WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : maxForward.getWidth(start) / 2);
420                    }
421                    if (placement.startsWith("right_of")) {
422                        offsetSoFar += Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (Utils.
423                                WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : maxForward.getWidth(start) / 2);
424                    }
425                }
426                valid = true;
427            }
428
429            if (!valid && direction == -1) {
430                if (minBackward._position+1 > lane) {
431                    int numLanesAway = minBackward._position + 1 - lane;
432                    offsetSoFar = minBackwardOffset + numLanesAway * Utils.WIDTH_LANES;
433
434                    if (placement.startsWith("left_of")) {
435                        offsetSoFar += Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (Utils.
436                                WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : minBackward.getWidth(start) / 2);
437                    }
438                    if (placement.startsWith("right_of")) {
439                        offsetSoFar -= Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ? (Utils.
440                                WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : minBackward.getWidth(start) / 2);
441                    }
442                }
443                if (maxBackward._position+1 < lane) {
444                    int numLanesAway = maxBackward._position + 1 - lane;
445                    offsetSoFar = maxBackwardOffset + numLanesAway * Utils.WIDTH_LANES;
446
447                    if (placement.startsWith("left_of")) {
448                        offsetSoFar -= Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ?
449                                (Utils.WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : maxBackward.getWidth(start) / 2);
450                    }
451                    if (placement.startsWith("right_of")) {
452                        offsetSoFar += Utils.RENDERING_WIDTH_DIVIDER / 2 + (ignoreWidthTags ?
453                                (Utils.WIDTH_LANES-Utils.RENDERING_WIDTH_DIVIDER)/2 : maxBackward.getWidth(start) / 2);
454                    }
455                }
456                valid = true;
457            }
458
459            if (!valid) return getWidth(start) / 2.0;
460
461            return offsetSoFar;
462        } catch (Exception e) {
463            return getWidth(start) / 2.0;
464        }
465    }

467    private String getPlacementTag(boolean start) {
468        String output = null;
469        if (start && _way.hasTag("placement:start") && _way.isOneway() == 1) {
470            output = _way.get("placement:start") + "f";
471        } else if (!start && _way.hasTag("placement:end") && _way.isOneway() == 1) {
472            output = _way.get("placement:end") + "f";
473        } else if (start && _way.hasTag("placement:forward:start")) {
474            output = _way.get("placement:forward:start") + "f";
475        } else if (!start && _way.hasTag("placement:forward:end")) {
476            output = _way.get("placement:forward:end") + "f";
477        } else if (start && _way.hasTag("placement:backward:start")) {
478            output = _way.get("placement:backward:start") + "b";
479        } else if (!start && _way.hasTag("placement:backward:end")) {
480            output = _way.get("placement:backward:end") + "b";
481        } else if (start && _way.hasTag("placement:both_ways:start")) {
482            output = _way.get("placement:both_ways:start") + "m";
483        } else if (!start && _way.hasTag("placement:both_ways:end")) {
484            output = _way.get("placement:both_ways:end") + "m";
485        } else if (_way.hasTag("placement") && _way.isOneway() == 1) {
486            output = _way.get("placement") + "f";
487        } else if (_way.hasTag("placement:forward")) {
488            output = _way.get("placement:forward") + "f";
489        } else if (_way.hasTag("placement:backward")) {
490            output = _way.get("placement:backward") + "b";
491        } else if (_way.hasTag("placement:both_ways")) {
492            output = _way.get("placement:both_ways") + "m";
493        }
494        return output;
495    }

497    private double getWidth(boolean start) {
498        double output = -1 * Utils.RENDERING_WIDTH_DIVIDER; // To offset the shoulder width added to each side.
499        for (RoadPiece piece : getRoadPieces(false)) {
500            output += piece.getWidth(start);
501        }
502        return output;

505    public List<RoadPiece> getRoadPieces(boolean renderingOrder) {
506        List<RoadPiece> output = new ArrayList<>();
507
508        if (renderingOrder) {
509            output.addAll(getLanesAndDividers(_forwardLanes, _forwardDividers, 1));
510            output.addAll(getLanesAndDividers(_backwardLanes, _backwardDividers, -1));
511            if (_backwardLanes.size() != 0 || (_bothWaysLane instanceof Lane)) {
512                output.add(_bothWaysLane);
513            }
514            output.add(_leftRoadEdge);
515            output.add(_rightRoadEdge);
516        } else {
517            output.add(_leftRoadEdge);
518            if (Utils.isRightHand(_way)) {
519                output.addAll(getLanesAndDividers(_backwardLanes, _backwardDividers, -1));
520                if (_backwardLanes.size() != 0 || _bothWaysLane instanceof Lane) output.add(_bothWaysLane);
521                output.addAll(getLanesAndDividers(_forwardLanes, _forwardDividers, 1));
522            } else {
523                output.addAll(getLanesAndDividers(_forwardLanes, _forwardDividers, 1));
524                if (_backwardLanes.size() != 0 || _bothWaysLane instanceof Lane) output.add(_bothWaysLane);
525                output.addAll(getLanesAndDividers(_backwardLanes, _backwardDividers, -1));
526            }
527            output.add(_rightRoadEdge);
528        }
529        return output;
530    } // TODO MAKE PRIVATE

532    private List<RoadPiece> getLanesAndDividers(List<RoadPiece> lanes, List<RoadPiece> dividers, int direction) {
533        List<RoadPiece> output = new ArrayList<>();
534        if (direction > 0) {
535            for (int i = 0; i < lanes.size(); i++) {
536                output.add(lanes.get(i));
537                if (i < lanes.size() - 1) output.add(dividers.get(i));
538            }
539        } else {
540            for (int i = lanes.size()-1; i >= 0; i--) {
541                output.add(lanes.get(i));
542                if (i != 0) output.add(dividers.get(i-1));
543            }
544        }
545        return output;
546    }

549    private void getOffsets() {
550        List<RoadPiece> roadPieces = getRoadPieces(false);
551        double offsetSoFarStart = _offsetToLeftStart;
552        double offsetSoFarEnd = _offsetToLeftEnd;
553        for (int i = 0; i < roadPieces.size(); i++) {
554            roadPieces.get(i).setOffset(offsetSoFarStart, offsetSoFarEnd);
555            offsetSoFarStart -= roadPieces.get(i).getWidth(true)/2 +
556                    (i != roadPieces.size()-1 ? roadPieces.get(i+1).getWidth(true)/2 : 0);
557            offsetSoFarEnd -= roadPieces.get(i).getWidth(false)/2 +
558                    (i != roadPieces.size()-1 ? roadPieces.get(i+1).getWidth(false)/2 : 0);
559        }
560    }

566    private void generateClickBoxes() {
567        Way leftEdge = Utils.getParallel(_alignment, _offsetToLeftStart, _offsetToLeftEnd, false);
568        Way rightEdge = Utils.getParallel(_alignment, _offsetToLeftStart - getWidth(true),
569                _offsetToLeftEnd - getWidth(false), false);
570        List<Node> outline = leftEdge.getNodes();
571        for (int i = rightEdge.getNodesCount()-1; i >= 0; i--) {
572            outline.add(rightEdge.getNode(i));
573        }
574        outline.add(leftEdge.getNode(0));
575        _outline = new Way();
576        _outline.setNodes(outline);
577    }

579    public boolean mouseEventIsInside(MouseEvent e) {
580        int[] xPoints = new int[_outline.getNodesCount()];
581        int[] yPoints = new int[xPoints.length];
582
583        for (int i = 0; i < _outline.getNodesCount(); i++) {
584            xPoints[i] = (int) (_mv.getPoint(_outline.getNode(i).getCoor()).getX() + 0.5);
585            yPoints[i] = (int) (_mv.getPoint(_outline.getNode(i).getCoor()).getY() + 0.5);
586        }
587
588        return new Polygon(xPoints, yPoints, xPoints.length).contains(e.getPoint());
589    }
